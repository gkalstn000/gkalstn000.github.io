import sys
sys.setrecursionlimit(10**6)

def solution(k, room_number):
    global pointer
    pointer = {}
    return [check_in(room) for room in room_number]

def check_in(room) :
    if room not in pointer :
        pointer[room] = room + 1
        return room
    pointer[room] = (avail := check_in(pointer[room]))
    return avail
