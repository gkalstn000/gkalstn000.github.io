```python
from itertools import accumulate,combinations
def solution(cookie):
    S = set(accumulate([0] + cookie))
    answer = [abs(x - y) // 2 for x, y in combinations(S, 2) if (x + y) % 2 == 0 and (x + y) // 2 in S]
    return max(answer, default=0)
```
