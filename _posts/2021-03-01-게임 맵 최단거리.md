왜 level4 인지 모르겠는 기본적인 `BFS` 문제.


```python
import numpy as np
def solution(maps):
    board = np.pad(maps, ((1,1),(1,1)), 'constant', constant_values=0).tolist()
    n, m = len(board)-2, len(board[0])-2
    d = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    que = [[1, 1, 1]]
    while que :
        h, w, cost = que.pop(0)
        if cost > n*m : continue
        if (h, w) == (n, m) : return cost
        for next_h, next_w in [[h+dh, w+dw] for dh, dw in d] :
            if board[next_h][next_w] != 0  :
                que.append([next_h, next_w, cost+1])
                board[next_h][next_w] = 0
    return -1
```
