from collections import deque
import bisect
def solution(food_times, k):
    answer = -1
    foods = deque(sorted((food, i) for i, food in enumerate(food_times)))
    remain = [1] * (total_food := len(food_times))
    cost_time = 0
    prev = 0
    while foods :
        n = len(foods)
        food, i = foods.popleft()
        cost_time += n * (food - prev)
        if cost_time  > k :
            remaining_foods = [k for k in range(total_food) if remain[k]]
            idx = (k - cost_time) % n
            answer = remaining_foods[idx] + 1
            break
        prev = food
        remain[i] = 0
    return answer
